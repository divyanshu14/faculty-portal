// create database
'use faculty' : Creates database faculty or swtiches to faculty db

// create collection
db.createCollection('profiles');

// insert document in collection
db.profiles.insert({name:"john doe",institute:"mit"});

// insert array of documents in database
db.profiles.insert([{name:"jane doe",institute:"iit",field:"math"},
    {name:"modi",institute:"bjp",field:"politics"}]);

// find docs in db
db.profiles.find()
db.profiles.find().pretty() // for well formatted output

// update in mongo
1) // completely replaces doc with first argument value by doc in the second argument
db.profiles.update({field:"cs"},{name:"john doe",institute:"mit",field:"cs"});
2) // using $set, we can just update particulars of the db
db.profiles.update({name:"modi"},{$set:{designation:"pm"}});
3) // unset the operation performed in the above
db.profiles.update({name:"modi"},{$unset:{designation:"pm"}});
